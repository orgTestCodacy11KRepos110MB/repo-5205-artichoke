Readline for Rust

This implementation is based on Antirez’s Linenoise

Example

Usage

let mut rl = rustyline::Editor::<()>::new()?;
let readline = rl.readline(">> ");
match readline {
    Ok(line) => println!("Line: {:?}", line),
    Err(_) => println!("No input"),
}
